    Iso.isoEq≋ iso
      (Seq.eq≋&ₚ (Seq.eq≋&ₚ it it)
      (Seq.eq≋&ₚ it
      (Seq.eq≋&ₚ it
      (Seq.eq≋&ₚ it
      (Seq.eq≋&ₚ it
      (Seq.eq≋&ₚ (Parallel.eq≋Σₚ it λ _ → record { _≟_ = λ where self self → yes refl })

      (Seq.eq≋&ₚ it it))))))))
    where
    instance
      e₁ : Eq≋ (NonEmptySequenceOf Extension)
