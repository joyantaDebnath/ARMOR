WSP = Sum (λ x → Erased (x ≡ [ ' ' ])) (λ x → Erased (x ≡ [ '\t' ]))

data EOL : @0 List Char → Set where
  crlf : EOL ('\r' ∷ [ '\n' ])
  cr   : EOL [ '\r' ]
  lf   : EOL [ '\n' ]
